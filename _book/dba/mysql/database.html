dev

###############################################################################
## 

CREATE DATABASE `dev` DEFAULT CHARACTER SET utf8 COLLATE utf8_general_ci;

## =============================================================================
## ChinaFuturesCalendar
CREATE TABLE IF NOT EXISTS dev.ChinaFuturesCalendar(
    nights Date,        ## 夜盘
    days Date NOT NULL    ## 日盘
)
